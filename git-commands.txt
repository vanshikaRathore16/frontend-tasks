//Config Command------------------------------------------------------------------------------
  The git config command in Git is used to set, get, and list configuration options

// Set global configuration
git config --global user.name "Your Name"
git config --global user.email "your.email@example.com"
 
// Set local configuration (for current repo)
git config user.name "Your Name"
git config user.email "your.email@example.com"

// View current config
git config user.name
git config user.email

// Clone a repository
git clone https://github.com/vanshikaRathore16/temprary

// List all branches
git branch

// Create a new branch
git branch button-update

// Create and switch to a new branch
git checkout -b indentation

// Add all files
git add .

// Commit changes with a message
git commit -m "change the indentation"

// Undo the last commit but keep the changes
git reset --soft HEAD~1

// Launch graphical repository viewer
gitk

// Push code to a remote branch
git push -u origin <branch_name>

// Save uncommitted changes (stash)
git stash

// List all stashed changes
git stash list

// Apply a specific stash without removing it
git stash apply stash@{1}

// Pop a specific stash (apply and remove it)
git stash pop stash@{2}

// Drop/delete a specific stash
git stash drop stash@{2}

// Initialize a git repository
git init

// Check status of working directory
git status

// Add remote origin
git remote add origin ""

// View remote URLs
git remote -v

// Add all files to staging area
git add .

// Commit with a message
git commit -m "msg"

// Push code to remote branch
git push -u origin <branch_name>

// Merge a branch into current branch
git merge ui-update

// Rename current branch
git branch -m <new_branch_name>

// Rename a different branch
git branch -m <old_branch_name> <new_branch_name>

// Push renamed branch
git push origin -u <new_branch_name>

// Delete a remote branch
git push origin --delete <old_branch_name>
